<script>
// ─────────────────────────────────────────
//  STATE

let playlist = [];
let currentIndex = -1;
let dragSrcIndex = null;

// ─────────────────────────────────────────
//  INIT
// ─────────────────────────────────────────
(function init() {
  const saved = localStorage.getItem('yuulist_data');
  if (saved) {
    try {
      const data = JSON.parse(saved);
      playlist = data.playlist || [];
      document.getElementById('pl-name').value = data.name || 'Mi Playlist';
    } catch(e) {}
  }
  renderList();

  document.getElementById('pl-name').addEventListener('input', save);
  document.getElementById('pl-name').addEventListener('keydown', e => {
    if (e.key === 'Enter') e.target.blur();
  });
})();

// ─────────────────────────────────────────
//  YOUTUBE UTILS
// ─────────────────────────────────────────
function extractVideoId(url) {
  url = url.trim();
  const patterns = [
    /(?:youtube\.com\/watch\?v=|youtu\.be\/|youtube\.com\/embed\/)([A-Za-z0-9_-]{11})/,
    /^([A-Za-z0-9_-]{11})$/
  ];
  for (const p of patterns) {
    const m = url.match(p);
    if (m) return m[1];
  }
  return null;
}

function thumbUrl(id) {
  return `https://img.youtube.com/vi/${id}/mqdefault.jpg`;
}

async function fetchTitle(id) {
  try {
    const r = await fetch(`https://www.youtube.com/oembed?url=https://www.youtube.com/watch?v=${id}&format=json`);
    if (!r.ok) throw new Error();
    const d = await r.json();
    return d.title || `Video ${id}`;
  } catch {
    return `Video ${id}`;
  }
}

// ─────────────────────────────────────────
//  ADD VIDEO
// ─────────────────────────────────────────
async function addVideo(url) {
  const id = extractVideoId(url);
  if (!id) { showToast('URL inválida ⚠'); return; }
  if (playlist.find(v => v.id === id)) { showToast('Video ya en la lista'); return; }

  const title = await fetchTitle(id);
  playlist.push({ id, title, fav: false });
  save();
  renderList();
  showToast('Video agregado ✓');
}

function addVideoFromInput() {
  const input = document.getElementById('url-input');
  const url = input.value.trim();
  if (!url) return;
  addVideo(url);
  input.value = '';
}

// ─────────────────────────────────────────
//  DRAG & DROP (drop zone for URL)
// ─────────────────────────────────────────
function handleDragOver(e) {
  e.preventDefault();
  document.getElementById('drop-zone').classList.add('drag-over');
}

function handleDragLeave() {
  document.getElementById('drop-zone').classList.remove('drag-over');
}

function handleDrop(e) {
  e.preventDefault();
  document.getElementById('drop-zone').classList.remove('drag-over');
  const url = e.dataTransfer.getData('text/plain') ||
              e.dataTransfer.getData('text/uri-list') ||
              e.dataTransfer.getData('URL');
  if (url) addVideo(url);
}

// ─────────────────────────────────────────
//  PLAY
// ─────────────────────────────────────────
function playVideo(index) {
  if (index < 0 || index >= playlist.length) return;
  currentIndex = index;
  const v = playlist[index];

  const iframe = document.getElementById('yt-player');
  const placeholder = document.getElementById('placeholder');
  iframe.style.display = 'block';
  placeholder.style.display = 'none';
  iframe.src = `https://www.youtube.com/embed/${v.id}?autoplay=1&rel=0`;

  document.getElementById('now-title').textContent = v.title;
  renderList();
}

function playAdjacentVideo(dir) {
  const next = currentIndex + dir;
  if (next >= 0 && next < playlist.length) playVideo(next);
}

// ─────────────────────────────────────────
//  RENDER
// ─────────────────────────────────────────
function renderList() {
  const wrap = document.getElementById('list-wrap');
  const empty = document.getElementById('empty-state');
  const meta = document.getElementById('pl-meta');

  const favCount = playlist.filter(v => v.fav).length;
  meta.textContent = `${playlist.length} video${playlist.length !== 1 ? 's' : ''}${favCount ? ` · ${favCount} ★` : ''}`;

  if (playlist.length === 0) {
    empty.style.display = 'flex';
    // remove all items
    [...wrap.querySelectorAll('.video-item')].forEach(el => el.remove());
    return;
  }
  empty.style.display = 'none';

  // Diff render (simple full re-render)
  [...wrap.querySelectorAll('.video-item')].forEach(el => el.remove());

  playlist.forEach((v, i) => {
    const item = document.createElement('div');
    item.className = 'video-item' + (i === currentIndex ? ' active' : '');
    item.dataset.index = i;
    item.draggable = true;

    item.innerHTML = `
      <div class="drag-handle" title="Arrastrar para reordenar">
        <svg width="14" height="14" viewBox="0 0 14 14" fill="none">
          <circle cx="5" cy="4" r="1" fill="currentColor"/>
          <circle cx="9" cy="4" r="1" fill="currentColor"/>
          <circle cx="5" cy="7" r="1" fill="currentColor"/>
          <circle cx="9" cy="7" r="1" fill="currentColor"/>
          <circle cx="5" cy="10" r="1" fill="currentColor"/>
          <circle cx="9" cy="10" r="1" fill="currentColor"/>
        </svg>
      </div>
      <img class="thumb" src="${thumbUrl(v.id)}" alt="" loading="lazy" onerror="this.style.opacity='.3'">
      <div class="item-info">
        <div class="item-title" title="${escHtml(v.title)}">${escHtml(v.title)}</div>
        <div class="item-id">${v.id}</div>
      </div>
      <div class="item-actions">
        <button class="icon-btn${v.fav ? ' fav' : ''}" title="${v.fav ? 'Quitar favorito' : 'Favorito'}"
                onclick="toggleFav(event,${i})">
          <svg width="14" height="14" viewBox="0 0 14 14" fill="${v.fav ? 'var(--fav)' : 'none'}">
            <path d="M7 1.5l1.5 3 3.3.5-2.4 2.3.6 3.3L7 9 4 10.6l.6-3.3L2.2 5l3.3-.5z"
                  stroke="${v.fav ? 'var(--fav)' : 'currentColor'}" stroke-width="1.2"
                  stroke-linejoin="round"/>
          </svg>
        </button>
        <button class="icon-btn del" title="Eliminar" onclick="removeVideo(event,${i})">
          <svg width="13" height="13" viewBox="0 0 14 14" fill="none">
            <path d="M2 4h10M5 4V2.5h4V4M11 4l-.7 7.5a1 1 0 01-1 .9H4.7a1 1 0 01-1-.9L3 4"
                  stroke="currentColor" stroke-width="1.3" stroke-linecap="round" stroke-linejoin="round"/>
          </svg>
        </button>
      </div>
    `;

    // Click to play (only on non-button areas)
    item.addEventListener('click', e => {
      if (!e.target.closest('button')) playVideo(i);
    });

    // ── List drag & drop reorder ──
    item.addEventListener('dragstart', e => {
      dragSrcIndex = i;
      item.classList.add('dragging');
      e.dataTransfer.effectAllowed = 'move';
      e.dataTransfer.setData('text/plain', String(i));
    });

    item.addEventListener('dragend', () => {
      item.classList.remove('dragging');
      wrap.querySelectorAll('.video-item').forEach(el => {
        el.classList.remove('drag-over-top', 'drag-over-bottom');
      });
    });

    item.addEventListener('dragover', e => {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'move';
      const rect = item.getBoundingClientRect();
      const mid = rect.top + rect.height / 2;
      item.classList.remove('drag-over-top', 'drag-over-bottom');
      item.classList.add(e.clientY < mid ? 'drag-over-top' : 'drag-over-bottom');
    });

    item.addEventListener('dragleave', () => {
      item.classList.remove('drag-over-top', 'drag-over-bottom');
    });

    item.addEventListener('drop', e => {
      e.preventDefault();
      item.classList.remove('drag-over-top', 'drag-over-bottom');
      if (dragSrcIndex === null || dragSrcIndex === i) return;

      const rect = item.getBoundingClientRect();
      const insertAfter = e.clientY >= rect.top + rect.height / 2;
      const [moved] = playlist.splice(dragSrcIndex, 1);
      let target = insertAfter ? i : i;
      if (dragSrcIndex < i && !insertAfter) target = i - 1;
      if (dragSrcIndex > i && insertAfter) target = i + 1;
      // simpler approach:
      playlist.splice(dragSrcIndex < i
        ? (insertAfter ? i : i - 1)
        : (insertAfter ? i + 1 : i), 0, moved);

      if (currentIndex === dragSrcIndex) {
        currentIndex = playlist.indexOf(moved);
      } else if (currentIndex !== -1) {
        // re-find current after mutation
        const cur = playlist[currentIndex];
        currentIndex = playlist.indexOf(cur);
      }

      dragSrcIndex = null;
      save();
      renderList();
    });

    wrap.appendChild(item);
  });
}

function escHtml(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// ─────────────────────────────────────────
//  ACTIONS
// ─────────────────────────────────────────
function toggleFav(e, index) {
  e.stopPropagation();
  playlist[index].fav = !playlist[index].fav;
  save();
  renderList();
}

function removeVideo(e, index) {
  e.stopPropagation();
  if (index === currentIndex) {
    // stop player
    const iframe = document.getElementById('yt-player');
    iframe.style.display = 'none';
    iframe.src = '';
    document.getElementById('placeholder').style.display = 'flex';
    document.getElementById('now-title').innerHTML = '<span class="now-empty">Ningún video seleccionado</span>';
    currentIndex = -1;
  } else if (index < currentIndex) {
    currentIndex--;
  }
  playlist.splice(index, 1);
  save();
  renderList();
  showToast('Video eliminado');
}

function clearPlaylist() {
  if (!playlist.length) return;
  if (!confirm('¿Limpiar toda la playlist?')) return;
  playlist = [];
  currentIndex = -1;
  const iframe = document.getElementById('yt-player');
  iframe.style.display = 'none';
  iframe.src = '';
  document.getElementById('placeholder').style.display = 'flex';
  document.getElementById('now-title').innerHTML = '<span class="now-empty">Ningún video seleccionado</span>';
  save();
  renderList();
  showToast('Playlist limpiada');
}

// ─────────────────────────────────────────
//  SAVE / LOAD
// ─────────────────────────────────────────
function save() {
  localStorage.setItem('yuulist_data', JSON.stringify({
    name: document.getElementById('pl-name').value,
    playlist
  }));
}

// ─────────────────────────────────────────
//  COPY / IMPORT
// ─────────────────────────────────────────
function openCopyModal() {
  const data = {
    name: document.getElementById('pl-name').value,
    playlist
  };
  document.getElementById('copy-textarea').value = JSON.stringify(data, null, 2);
  document.getElementById('copy-modal').classList.add('open');
}

function copyToClipboard() {
  const ta = document.getElementById('copy-textarea');
  ta.select();
  navigator.clipboard.writeText(ta.value)
    .then(() => showToast('¡Copiado al portapapeles! ✓'))
    .catch(() => { document.execCommand('copy'); showToast('¡Copiado! ✓'); });
  closeModal('copy-modal');
}

function openImportModal() {
  document.getElementById('import-textarea').value = '';
  document.getElementById('import-modal').classList.add('open');
}

function importPlaylist() {
  const text = document.getElementById('import-textarea').value.trim();
  if (!text) return;
  try {
    const data = JSON.parse(text);
    if (!Array.isArray(data.playlist)) throw new Error();
    playlist = data.playlist;
    if (data.name) document.getElementById('pl-name').value = data.name;
    currentIndex = -1;
    const iframe = document.getElementById('yt-player');
    iframe.style.display = 'none';
    iframe.src = '';
    document.getElementById('placeholder').style.display = 'flex';
    document.getElementById('now-title').innerHTML = '<span class="now-empty">Ningún video seleccionado</span>';
    save();
    renderList();
    closeModal('import-modal');
    showToast(`Playlist importada: ${playlist.length} videos ✓`);
  } catch {
    showToast('Formato inválido ⚠');
  }
}

function closeModal(id) {
  document.getElementById(id).classList.remove('open');
}

// Close modals on overlay click
document.querySelectorAll('.modal-overlay').forEach(overlay => {
  overlay.addEventListener('click', e => {
    if (e.target === overlay) overlay.classList.remove('open');
  });
});

// ─────────────────────────────────────────
//  TOAST
// ─────────────────────────────────────────
let toastTimer;
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => t.classList.remove('show'), 2200);
}
